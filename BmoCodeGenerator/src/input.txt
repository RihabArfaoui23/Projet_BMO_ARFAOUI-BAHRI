    abstract class Utilisateur {
        private userId : Int
        private email : String
        private password : String

    }

        private capitalInitial: Float 
        private jetons: Float

        public determinerCotes(): void
        public determinerEcartsScores(): void

        private cote: Float

        public calculerGain(): Float
        public consulterResultat(): String

    }

        private idEvenement: Int

    }

    }
